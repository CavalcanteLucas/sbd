&\hspace{30pt}\text{\textbf{if} $\beta\subseteq result$ \textbf{then} $result := result\cup\gamma$;} \\
&\hspace{20pt}\text{\textbf{end}} \\
&\text{\textbf{until}(\textit{result} não mudar)} \\
\end{aligned}
$$
